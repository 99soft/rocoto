16  package org.nnsoft.guice.rocoto.converters;
17  
18  import java.net.MalformedURLException;
19  import java.net.URL;
20  
21  import com.google.inject.ProvisionException;
22  import com.google.inject.TypeLiteral;
23  
24  /**
25   * Converter implementation for {@code java.net.URL}.
26   */
27  public final class URLConverter
28      extends AbstractConverter<URL>
29  {
30  
31      /**
32       * Pseudo URL prefix for loading from the class path: "classpath://"
33       */
34      public static final String CLASSPATH_URL_PREFIX = "classpath://";
35  
36      /**
37       * {@inheritDoc}
38       */
39      public Object convert( String value, TypeLiteral<?> toType )
40      {
41          if ( value.startsWith( CLASSPATH_URL_PREFIX ) )
42          {
43              String path = value.substring( CLASSPATH_URL_PREFIX.length() );
44              while ( '/' == path.charAt( 0 ) )
45              {
46                  path = path.substring( 1 );
47              }
48  
49              ClassLoader classLoader = null;
50              try
51              {
52                  classLoader = Thread.currentThread().getContextClassLoader();
53              }
54              catch ( Throwable t )
55              {
56                  // Cannot access thread context ClassLoader - falling back to system class loader...
57              }
58              if ( classLoader == null )
59              {
60                  // No thread context class loader -> use class loader of this class.
61                  classLoader = URLConverter.class.getClassLoader();
62              }
63              URL url = classLoader.getResource( path );
64              if ( url == null )
65              {
66                  throw new ProvisionException( "class path resource '"
67                                                + path
68                                                + "' cannot be resolved to URL because it does not exist" );
69              }
70  
71              return url;
72          }
73  
74          try
75          {
76              return new URL( value );
77          }
78          catch ( MalformedURLException e )
79          {
80              throw new ProvisionException( "String value '" + value + "' is not a valid URL", e );
81          }
82      }
83  
84  }
