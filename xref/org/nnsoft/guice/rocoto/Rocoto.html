18  import static com.google.inject.name.Names.named;
19  import static com.google.inject.spi.Elements.getElements;
20  import static com.google.inject.util.Modules.override;
21  import static java.util.Arrays.asList;
22  
23  import java.util.List;
24  import java.util.Map.Entry;

26  import org.nnsoft.guice.rocoto.variables.VariablesMap;
27  
28  import com.google.inject.AbstractModule;
29  import com.google.inject.Binding;
30  import com.google.inject.Key;
31  import com.google.inject.Module;
32  import com.google.inject.TypeLiteral;
33  import com.google.inject.name.Named;
34  import com.google.inject.spi.DefaultElementVisitor;
35  import com.google.inject.spi.Element;
36  
37  /**
38   * @since 6.0
39   */
40  public final class Rocoto
41      extends AbstractModule
42  {
43  
44      public static Module expandVariables( Module...baseModules )
45      {
46          return expandVariables( asList( baseModules ) );
47      }
48  
49      public static Module expandVariables( Iterable<? extends Module> baseModules )
50      {
51          return override( baseModules ).with( new Rocoto( getElements( baseModules ) ) );
52      }
53  
54      private final TypeLiteral<String> stringLiteral = new TypeLiteral<String>(){};
55  
56      private final List<Element> elements;
57  
58      /**
59       * Do nothing, this class cannot be instantiated
60       */
61      private Rocoto( List<Element> elements )
62      {
63          this.elements = elements;
64      }
65  
66      @Override
67      protected void configure()
68      {
69          final VariablesMap variablesMap = new VariablesMap();
70  
71          for ( final Element element : elements )
72          {
73              element.acceptVisitor( new DefaultElementVisitor<Void>()
74              {
75  
76                  @Override
77                  public <T> Void visit( Binding<T> binding )
78                  {
79                      Key<?> bindingKey = binding.getKey();
80  
81                      if ( stringLiteral.equals( bindingKey.getTypeLiteral() )
82                              && bindingKey.getAnnotation() != null
83                              && ( Named.class.isAssignableFrom( bindingKey.getAnnotationType() )
84                                              || javax.inject.Named.class.isAssignableFrom( bindingKey.getAnnotationType() ) ) )
85                      {
86                          String propertyKey;
87  
88                          if ( Named.class.isAssignableFrom( bindingKey.getAnnotationType() ) )
89                          {
90                              propertyKey = ( (Named) bindingKey.getAnnotation() ).value();
91                          }
92                          else
93                          {
94                              propertyKey = ( (javax.inject.Named) bindingKey.getAnnotation() ).value();
95                          }
96  
97                          String propertyValue = (String) binding.getProvider().get();
98  
99                          variablesMap.put( propertyKey, propertyValue );
100                     }
101 
102                     return super.visit( binding );
103                 }
104 
105             });
106         }
107 
108         for ( Entry<String, String> variable : variablesMap.entrySet() )
109         {
110             bindConstant().annotatedWith( named( variable.getKey() ) ).to( variable.getValue() );
111         }
112     }
113 
114 }
